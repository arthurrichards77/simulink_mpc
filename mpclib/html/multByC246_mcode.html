589function y = multByC(z,A,B,Ef,T,n,m,ellef)
590%
591% return C*z
592%
593
594y = zeros(T*n+ellef,1);
595
596y(1:n) = z(m+(1:n)) - B*z(1:m);
597for ii=2:T,
598    y(n*(ii-1)+(1:n)) = z(m+(ii-1)*(n+m)+(1:n)) - A*z(m+(ii-2)*(n+m)+(1:n)) - B*z((ii-1)*(n+m)+(1:m));
599end
600
601if ellef>0,
602    y(T*n + (1:ellef)) = Ef*z(m+(T-1)*(n+m)+(1:n));
603end
604
605end
