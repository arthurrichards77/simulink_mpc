607function y = multByP(z,Fx,Fu,Ff,T,n,m,ell,ellf)
608%
609% return P*z
610%
611
612y = zeros(T*ell+ellf,1);
613
614y(1:ell) = Fu*z(1:m);
615for ii=2:T,
616    y(ell*(ii-1)+(1:ell)) = Fx*z(m+(ii-2)*(n+m)+(1:n)) + Fu*z((ii-1)*(n+m)+(1:m));
617end
618if ellf>0,
619    y(T*ell + (1:ellf)) = Ff*z(m+(T-1)*(n+m)+(1:n));
620end
621
622end
