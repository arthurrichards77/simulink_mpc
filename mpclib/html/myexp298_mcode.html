564function y = myexp(u,flagNoPade)
565
566% use built in exponential
567if flagNoPade,
568    y = exp(u);
569else
570    % use pade approximant
571    di = ones(size(u));
572    ds = di;
573    for ii=1:3,
574        di = (-1/ii)*di.*u;
575        ds = ds+di;
576    end
577    y = 1./ds;
578end
579
580end
